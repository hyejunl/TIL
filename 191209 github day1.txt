3.ls                      폴더가 뭐가 있냐

 cd test/                test라는 폴더 안으로 들어가라
 cd ..                    test라는 폴더에서 빠져나와라
 git init                 git 저장소를 만들어라 -> .git 폴더가 생김
 touch a.txt            a.txt라는 파일을 만들게 됨
 git status             이력을 확인하는 것
 git add a.txt
 git status
 git commit -m 'a.txt 추가'
